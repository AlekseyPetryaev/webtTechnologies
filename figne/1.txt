# в первой строчке мы обозначаем в каком фармате будет файл 
<!DOCTYPE htm1> 
# здесь мы открываем наш файл
<html>
# обьевляем в голове
<head>
# задаем язык (русский)
<meta charset="UTF-8">
# Задаем название нашей страницы
# <тег открывает> Моя первая страница</тег закрывает>
<title>Моя первая страница</title>
# головной закрывающийся тег 
</head>
# указываем что будет на странице
<body>
# указываем какого уровня будет заголовок 
        <h1>Это заголовок первого уровня</h1>
        # Обзац мы ставим через 
        <p>Какой-то текст</p>
</body>
# этот тэг закрывает наш файл
</html>
